# 第 12 课：语句节点体系详解

## 语句节点概览

Druid 支持 284 种语句节点（`ast/statement/` 目录），覆盖了绝大多数数据库的 DML、DDL、DCL 语句。

## DML 语句（四大金刚）

### 1. SQLSelectStatement — 查询语句

```java
// SQLSelectStatement.java
public class SQLSelectStatement extends SQLStatementImpl {
    private SQLSelect select;

    // getSelect() → SQLSelect
    //   ├── getQuery() → SQLSelectQuery
    //   │   ├── SQLSelectQueryBlock (普通 SELECT)
    //   │   │   ├── getSelectList() → List<SQLSelectItem>
    //   │   │   ├── getFrom() → SQLTableSource
    //   │   │   ├── getWhere() → SQLExpr
    //   │   │   ├── getGroupBy() → SQLSelectGroupByClause
    //   │   │   ├── getOrderBy() → SQLOrderBy
    //   │   │   └── getLimit() → SQLLimit
    //   │   └── SQLUnionQuery (UNION)
    //   │       ├── getLeft() → SQLSelectQuery
    //   │       └── getRight() → SQLSelectQuery
    //   ├── getWithSubQuery() → SQLWithSubqueryClause (CTE)
    //   └── getOrderBy() → SQLOrderBy
    //   └── getLimit() → SQLLimit
}
```

### 2. SQLInsertStatement — 插入语句

```java
// SQLInsertStatement.java
public class SQLInsertStatement extends SQLStatementImpl {
    private SQLName tableName;           // INSERT INTO table
    private List<SQLExpr> columns;       // (col1, col2, ...)
    private ValuesClause valuesList;     // VALUES (v1, v2)
    private SQLSelect query;             // SELECT 子查询
    private List<SQLCommentHint> hints;  // INSERT 提示

    // 内部类 ValuesClause
    public class ValuesClause extends SQLObjectImpl {
        private List<SQLExpr> values;    // (v1, v2, ...)
        private List<SQLInsertStatement> mergedInsertList; // 批量插入
    }
}
```

### 3. SQLUpdateStatement — 更新语句

```java
// SQLUpdateStatement.java
public class SQLUpdateStatement extends SQLStatementImpl {
    private SQLTableSource tableSource;      // UPDATE table
    private List<SQLUpdateSetItem> items;    // SET col = value
    private SQLExpr where;                   // WHERE condition

    // SQLUpdateSetItem
    public class SQLUpdateSetItem extends SQLObjectImpl {
        private SQLExpr column;   // 列名
        private SQLExpr value;    // 值
    }
}
```

### 4. SQLDeleteStatement — 删除语句

```java
// SQLDeleteStatement.java
public class SQLDeleteStatement extends SQLStatementImpl {
    private SQLTableSource tableSource;    // DELETE FROM table
    private SQLExpr where;                 // WHERE condition
}
```

## DDL 语句样例

### SQLCreateTableStatement

```java
public class SQLCreateTableStatement extends SQLStatementImpl {
    private SQLName name;                  // 表名
    private List<SQLTableElement> tableElementList;  // 列和约束定义
    private List<SQLName> like;            // LIKE 子句
    private SQLExprTableSource inherits;   // INHERITS (PG)
    private SQLSelect query;               // CREATE TABLE AS SELECT
    private List<SQLAssignItem> tableOptions;  // ENGINE, CHARSET 等

    // SQLColumnDefinition (列定义)
    public class SQLColumnDefinition extends SQLObjectImpl {
        private SQLName name;              // 列名
        private SQLDataType dataType;      // 数据类型
        private SQLExpr defaultExpr;       // DEFAULT 值
        private List<SQLColumnConstraint> constraints;  // 约束
    }
}
```

## 获取语句信息的完整示例

```java
public class StatementInspector {
    public static void inspect(String sql, DbType dbType) {
        List<SQLStatement> stmts = SQLUtils.parseStatements(sql, dbType);

        for (SQLStatement stmt : stmts) {
            System.out.println("=== 语句类型: " + stmt.getClass().getSimpleName() + " ===");

            if (stmt instanceof SQLSelectStatement) {
                inspectSelect((SQLSelectStatement) stmt);
            } else if (stmt instanceof SQLInsertStatement) {
                inspectInsert((SQLInsertStatement) stmt);
            } else if (stmt instanceof SQLUpdateStatement) {
                inspectUpdate((SQLUpdateStatement) stmt);
            } else if (stmt instanceof SQLDeleteStatement) {
                inspectDelete((SQLDeleteStatement) stmt);
            }
        }
    }

    private static void inspectSelect(SQLSelectStatement stmt) {
        SQLSelectQuery query = stmt.getSelect().getQuery();

        if (query instanceof SQLSelectQueryBlock) {
            SQLSelectQueryBlock block = (SQLSelectQueryBlock) query;
            System.out.println("  SELECT 列: " + block.getSelectList().size());
            for (SQLSelectItem item : block.getSelectList()) {
                System.out.println("    - " + item.getExpr());
            }

            SQLTableSource from = block.getFrom();
            if (from != null) {
                System.out.println("  FROM: " + from);
                if (from instanceof SQLJoinTableSource) {
                    SQLJoinTableSource join = (SQLJoinTableSource) from;
                    System.out.println("    JOIN 类型: " + join.getJoinType());
                    System.out.println("    ON 条件: " + join.getCondition());
                }
            }

            if (block.getWhere() != null) {
                System.out.println("  WHERE: " + block.getWhere());
            }
            if (block.getOrderBy() != null) {
                System.out.println("  ORDER BY: " + block.getOrderBy());
            }
            if (block.getLimit() != null) {
                System.out.println("  LIMIT: " + block.getLimit());
            }
        }
    }

    private static void inspectInsert(SQLInsertStatement stmt) {
        System.out.println("  INTO 表: " + stmt.getTableName());
        System.out.println("  列: " + stmt.getColumns());
        if (stmt.getValues() != null) {
            System.out.println("  值: " + stmt.getValues().getValues());
        }
    }

    private static void inspectUpdate(SQLUpdateStatement stmt) {
        System.out.println("  更新表: " + stmt.getTableSource());
        System.out.println("  SET 项:");
        for (SQLUpdateSetItem item : stmt.getItems()) {
            System.out.println("    " + item.getColumn() + " = " + item.getValue());
        }
        System.out.println("  WHERE: " + stmt.getWhere());
    }

    private static void inspectDelete(SQLDeleteStatement stmt) {
        System.out.println("  删除表: " + stmt.getTableSource());
        System.out.println("  WHERE: " + stmt.getWhere());
    }
}
```

## SchemaStatVisitor — 自动收集信息

Druid 提供了 `SchemaStatVisitor`，可以自动遍历 AST 并收集表、列、条件等信息：

```java
String sql = "SELECT u.id, u.name FROM users u JOIN orders o ON u.id = o.user_id WHERE u.age > 18";

SchemaStatVisitor visitor = SQLUtils.createSchemaStatVisitor(DbType.mysql);
SQLUtils.parseStatements(sql, DbType.mysql).get(0).accept(visitor);

System.out.println("涉及的表:");
visitor.getTables().forEach((name, stat) -> {
    System.out.println("  " + name + " (操作: " + stat.getMode() + ")");
});

System.out.println("涉及的列:");
visitor.getColumns().forEach(col -> {
    System.out.println("  " + col.getTable() + "." + col.getName());
});

System.out.println("条件:");
visitor.getConditions().forEach(cond -> {
    System.out.println("  " + cond.getColumn() + " " + cond.getOperator() + " " + cond.getValues());
});
```

## 💡 SQL-to-ES DSL 实战思考

基于语句节点的结构，你的 SQL to ES DSL 转换器需要处理的典型场景：

```java
// SELECT → ES search
SQLSelectStatement → ES SearchRequest
  ├── selectList → _source (fetch fields)
  ├── where → query (bool/filter/term/range)
  ├── orderBy → sort
  ├── limit → size
  └── tableSource → index name

// INSERT → ES index
SQLInsertStatement → ES IndexRequest
  ├── tableName → index name
  └── values → document fields

// UPDATE → ES update
SQLUpdateStatement → ES UpdateRequest
  ├── tableSource → index name
  ├── items → doc fields
  └── where → query

// DELETE → ES delete
SQLDeleteStatement → ES DeleteByQueryRequest
  ├── tableSource → index name
  └── where → query
```
